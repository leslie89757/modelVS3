# ModelVS3 环境变量配置模板
# 将此内容复制到 .env 文件中并修改相应的值

# 数据库配置
POSTGRES_PASSWORD=your_secure_password
SECRET_KEY=your_secret_key_here
JWT_SECRET_KEY=your_jwt_secret_key_here

# API Keys
OPENAI_API_KEY=your_openai_api_key
ANTHROPIC_API_KEY=your_anthropic_api_key
GOOGLE_API_KEY=your_google_api_key

# CORS 配置
CORS_ORIGINS=http://localhost:3003,https://yourdomain.com

# 日志级别
LOG_LEVEL=INFO

# 监控配置
GF_SECURITY_ADMIN_PASSWORD=admin

# 前端API地址
VITE_API_URL=https://yourdomain.com:8000

# ==========================================
# 公网域名配置 - 用于二维码和分享链接
# ==========================================

# 公网域名或IP地址（必填）
# 示例: yourdomain.com 或 123.456.789.101
PUBLIC_DOMAIN=yourdomain.com

# 公网访问端口（可选）
# 如果不是标准端口（80/443），请指定
# 示例: 3003, 8080, 443
PUBLIC_PORT=443

# 协议类型（可选，默认https）
# 可选值: http, https
PUBLIC_SCHEME=https

# 配置示例：
# 1. 使用域名 + HTTPS（推荐）：
#    PUBLIC_DOMAIN=myapp.example.com
#    PUBLIC_PORT=443
#    PUBLIC_SCHEME=https
#    结果: https://myapp.example.com/experience/{agent_id}
#
# 2. 使用域名 + 自定义端口：
#    PUBLIC_DOMAIN=myapp.example.com
#    PUBLIC_PORT=8080
#    PUBLIC_SCHEME=https
#    结果: https://myapp.example.com:8080/experience/{agent_id}
#
# 3. 使用IP地址：
#    PUBLIC_DOMAIN=123.456.789.101
#    PUBLIC_PORT=3003
#    PUBLIC_SCHEME=http
#    结果: http://123.456.789.101:3003/experience/{agent_id}
